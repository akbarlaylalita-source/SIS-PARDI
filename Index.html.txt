let tempTotpToken = "";

function handleLogin(e) {
    e.preventDefault();
    const u = document.getElementById('username').value;
    const p = document.getElementById('password').value;
    const btn = document.getElementById('btnLogin');
    const err = document.getElementById('loginError');

    btn.innerHTML = `<div class="loader"></div>`;
    btn.disabled = true;
    err.classList.add('hidden');

    google.script.run.withSuccessHandler(res => {
        if (res.success && res.require2FA) {
            tempTotpToken = res.tempToken;
            document.getElementById('form-login-step1').classList.add('hidden');
            document.getElementById('form-login-step2').classList.remove('hidden');
            
            if (res.isSetup) {
                document.getElementById('totp-setup-container').classList.remove('hidden');
                document.getElementById('totp-setup-container').classList.add('flex');
                document.getElementById('totp-qr').src = res.qrUrl;
            } else {
                document.getElementById('totp-setup-container').classList.add('hidden');
                document.getElementById('totp-setup-container').classList.remove('flex');
            }
            
            btn.innerHTML = `<span>MASUK APLIKASI</span><i class="ph ph-arrow-right font-bold"></i>`;
            btn.disabled = false;
        } else {
            err.innerText = res.message || "Username atau password salah!";
            err.classList.remove('hidden');
            btn.innerHTML = `<span>MASUK APLIKASI</span><i class="ph ph-arrow-right font-bold"></i>`;
            btn.disabled = false;
        }
    }).loginUser(u, p);
}

function handleTOTP(e) {
    e.preventDefault();
    const code = document.getElementById('totp-code').value;
    const btn = document.getElementById('btnTOTP');
    const err = document.getElementById('loginError');

    btn.innerHTML = `<div class="loader"></div>`;
    btn.disabled = true;
    err.classList.add('hidden');

    google.script.run.withSuccessHandler(res => {
        if (res.success) {
            const userData = {
                username: res.username,
                nama: res.nama,
                role: res.role,
                token: res.token
            };

            localStorage.setItem('sispardi_user', JSON.stringify(userData));
            localStorage.setItem('sispardi_token', res.token);

            if(res.role === 'Admin' && document.getElementById('menuBackup')) {
                document.getElementById('menuBackup').classList.remove('hidden');
            }

            showApp(userData);
            
            btn.innerHTML = `<span>VERIFIKASI KODE</span><i class="ph ph-check-circle font-bold"></i>`;
            btn.disabled = false;
            document.getElementById('totp-code').value = "";
            
        } else {
            err.innerText = res.message || "Kode Authenticator salah!";
            err.classList.remove('hidden');
            btn.innerHTML = `<span>VERIFIKASI KODE</span><i class="ph ph-check-circle font-bold"></i>`;
            btn.disabled = false;
            document.getElementById('totp-code').value = "";
        }
    }).verify2FA(tempTotpToken, code);
}

function filterTableAktif() {
    // 1. Ambil Value dari Input
    const k = document.getElementById('searchKodeAktif').value.toLowerCase();
    const n = document.getElementById('searchNamaAktif').value.toLowerCase();
    
    // Ambil input tanggal (Cek elemen dulu biar tidak error)
    const elmStart = document.getElementById('dateStartAktif');
    const elmEnd = document.getElementById('dateEndAktif');
    const startDateVal = elmStart ? elmStart.value : "";
    const endDateVal = elmEnd ? elmEnd.value : "";

    // 2. Filter Data (Gunakan 'masterAktifData' bukan 'arsipAktifData')
    filteredAktifData = masterAktifData.filter(i => {
        // A. Filter Teks (Kode, Uraian, NoBerkas)
        const textKode = (i.KodeKlasifikasi || '').toLowerCase();
        const textUraian = (i.Uraian || '').toLowerCase();
        const textNoBerkas = (i.NoBerkas || '').toLowerCase();

        const matchText = textKode.includes(k) && (textUraian.includes(n) || textNoBerkas.includes(n));
        
        // B. Filter Tanggal (Hanya jalan jika user mengisi kedua tanggal)
        let matchDate = true;
        if (startDateVal && endDateVal && i.TglInput) {
            // Set jam ke 00:00:00 agar perbandingan akurat
            const itemDate = new Date(i.TglInput).setHours(0,0,0,0);
            const start = new Date(startDateVal).setHours(0,0,0,0);
            const end = new Date(endDateVal).setHours(0,0,0,0);
            
            // Cek apakah tanggal arsip berada di antara rentang
            matchDate = itemDate >= start && itemDate <= end;
        }

        return matchText && matchDate;
    });
    
    // 3. Render Ulang Tabel & Reset ke Halaman 1
    pageAktif = 1;
    renderPaginationAktif(); // Pastikan fungsi render pagination Anda dipanggil
}

function usulPindahInaktif() {
    // Logika untuk memfilter data yang 'sudah kadaluarsa' (Usia > Retensi Aktif)
    const candidates = masterAktifData.filter(row => {
        const rawDate = row.Tahun || row.TglInput;
        if(!rawDate) return false;
        
        let fileYear;
        if (rawDate instanceof Date) {
            fileYear = rawDate.getFullYear();
        } else {
            fileYear = parseInt(String(rawDate).substring(0, 4));
        }

        const currentYear = new Date().getFullYear();
        const retensi = parseInt(row.RetensiAktif) || 0;
        
        return (currentYear - fileYear) >= retensi;
    });

    if(candidates.length > 0) {
        Swal.fire({
            title: 'Usulan Pemindahan',
            html: `Ditemukan <b>${candidates.length} berkas</b> yang masa retensi aktifnya sudah habis.<br/><br/><span class="text-sm text-slate-500">Fitur pemindahan otomatis akan diproses pada menu batch action versi berikutnya.</span>`,
            icon: 'success',
            confirmButtonText: 'Mengerti',
            confirmButtonColor: '#3b82f6',
            heightAuto: false, // <--- TAMBAHKAN INI (PENTING)
            customClass: {
                popup: 'rounded-2xl',
                confirmButton: 'rounded-xl px-6'
            }
        });
    } else {
        Swal.fire({
            title: 'Tidak Ada Berkas',
            text: "Belum ada berkas yang memenuhi syarat untuk dipindahkan ke Inaktif (Masa Retensi Aktif belum habis).",
            icon: 'info',
            confirmButtonText: 'Tutup',
            confirmButtonColor: '#64748b',
            heightAuto: false, // <--- TAMBAHKAN INI (PENTING)
            customClass: {
                popup: 'rounded-2xl',
                confirmButton: 'rounded-xl px-6'
            }
        });
    }
}

function renderPaginationAktif() {
    const tbody = document.getElementById('tableAktifBody');
    const limit = parseInt(document.getElementById('limitAktif').value);
    const totalData = filteredAktifData.length;
    
    // Update Label Total
    document.getElementById('countAktif').innerText = totalData;

    // Hitung Max Page
    const maxPage = Math.ceil(totalData / limit);
    if (pageAktif < 1) pageAktif = 1;
    if (pageAktif > maxPage && maxPage > 0) pageAktif = maxPage;

    // Slicing Data
    const startIdx = (pageAktif - 1) * limit;
    const endIdx = Math.min(startIdx + limit, totalData);
    const displayData = filteredAktifData.slice(startIdx, endIdx);

    // Render HTML
    if (displayData.length === 0) {
        tbody.innerHTML = '<tr><td colspan="14" class="p-8 text-center text-slate-400 italic">Data tidak ditemukan.</td></tr>';
        document.getElementById('infoStartAktif').innerText = 0;
        document.getElementById('infoEndAktif').innerText = 0;
    } else {
        // Gunakan fungsi createRow yang sudah ada (pastikan createRow menerima index global jika perlu, tapi index halaman cukup)
        // Kita ubah idx di createRow agar nomor urut sesuai halaman (startIdx + i)
        tbody.innerHTML = displayData.map((r, i) => createRow(r, startIdx + i, true)).join('');
        document.getElementById('infoStartAktif').innerText = startIdx + 1;
        document.getElementById('infoEndAktif').innerText = endIdx;
    }

    // Update Tombol Navigasi
    document.getElementById('pageLabelAktif').innerText = `Page ${totalData === 0 ? 0 : pageAktif} / ${maxPage}`;
    document.getElementById('btnPrevAktif').disabled = (pageAktif <= 1);
    document.getElementById('btnNextAktif').disabled = (pageAktif >= maxPage || totalData === 0);
}

function loadKlasifikasi() {
    // 1. Tampilkan animasi loading di tabel
    const tbody = document.getElementById('tableKlasifikasiBody');
    if(tbody) {
        tbody.innerHTML = `<tr><td colspan="3" class="p-8 text-center text-slate-400"><div class="loader mx-auto mb-2"></div>Memuat data...</td></tr>`;
    }

    // 2. Panggil Backend Google Apps Script
    google.script.run.withSuccessHandler(data => {
        
        // [PENTING] Simpan data mentah ke variabel global untuk client-side processing
        rawKlasifikasiData = data;
        
        // 3. Update Datalist (Autocomplete) di Modal
        // Kita menggunakan string concatenation innerHTML sekaligus agar rendering browser Cepat
        const dataList = document.getElementById('listKodeOptions');
        if (dataList) {
            let optionsHTML = "";
            
            // Sortir data agar dropdown rapi (000, 000.1, dst)
            let sortedData = [...data].sort((a,b) => a.kode.localeCompare(b.kode, undefined, {numeric:true, sensitivity:'base'}));
            
            sortedData.forEach(item => {
                // VALIDASI: Hanya masukkan ke list jika kode TIDAK kosong
                if(item.kode && String(item.kode).trim() !== "") {
                    // Gunakan format: Value=Kode, Label=Deskripsi (Agar user bisa ketik kode)
                    optionsHTML += `<option value="${item.kode}">${item.deskripsi}</option>`;
                }
            });
            
            // Masukkan ke DOM sekali saja (Batch update) -> Mencegah Lag/Berat
            dataList.innerHTML = optionsHTML;
        }

        // 4. Render Tabel & Reset ke Halaman 1
        // Pastikan fungsi resetPageAndRender sudah ada (dari langkah sebelumnya)
        if (typeof resetPageAndRender === "function") {
            resetPageAndRender();
        } else {
            // Fallback jika fungsi reset belum didefinisikan
            renderFilteredKlasifikasi();
        }

    }).getKlasifikasi();
}

function resetPageAndRender() {
    currentPage = 1; // Reset ke halaman 1
    renderFilteredKlasifikasi();
}

    currentPage += direction; // Tambah atau Kurang halaman
    renderFilteredKlasifikasi();
}

// FUNGSI BARU: Render dengan Filter & Limit
// ==========================================
function renderFilteredKlasifikasi() {
    const tbody = document.getElementById('tableKlasifikasiBody');
    const searchInput = document.getElementById('searchKlasifikasi');
    const limitInput = document.getElementById('limitKlasifikasi');
    
    // Elemen UI Pagination (Footer)
    const btnPrev = document.getElementById('btnPrev');
    const btnNext = document.getElementById('btnNext');
    const pageInd = document.getElementById('pageIndicator');
    const infoStart = document.getElementById('pageInfoStart');
    const infoEnd = document.getElementById('pageInfoEnd');
    const infoTotal = document.getElementById('pageInfoTotal');
    
    // Safety check: Pastikan elemen ada sebelum lanjut
    if (!searchInput || !limitInput) return;

    const searchVal = searchInput.value.toLowerCase();
    const limitVal = parseInt(limitInput.value);
    
    // --- STEP A: FILTERING DATA ---
    let filtered = rawKlasifikasiData.filter(item => {
        // 1. Validasi: Buang baris jika Kodenya kosong/spasi saja
        const kodeAda = item.kode && String(item.kode).trim() !== "";
        if (!kodeAda) return false;

        // 2. Pencarian: Cek apakah Kode ATAU Deskripsi mengandung teks pencarian
        return (item.kode || '').toLowerCase().includes(searchVal) || 
               (item.deskripsi || '').toLowerCase().includes(searchVal);
    });

    // --- STEP B: SORTING ---
    // Mengurutkan secara natural (0, 000.1, 100, 200)
    filtered.sort((a,b) => a.kode.localeCompare(b.kode, undefined, {numeric:true, sensitivity:'base'}));

    // --- STEP C: KALKULASI PAGINATION ---
    const totalData = filtered.length;
    const totalPages = Math.ceil(totalData / limitVal);

    // Validasi Current Page (Mencegah error jika halaman > total halaman)
    if (currentPage < 1) currentPage = 1;
    if (currentPage > totalPages && totalPages > 0) currentPage = totalPages;

    // --- STEP D: UPDATE UI JIKA KOSONG ---
    if (totalData === 0) {
        tbody.innerHTML = `
            <tr>
                <td colspan="3" class="p-8 text-center text-slate-400 flex flex-col items-center gap-2">
                    <i class="ph ph-warning-circle text-3xl opacity-50"></i>
                    <span>Data tidak ditemukan.</span>
                </td>
            </tr>`;
        
        // Reset Info Footer jadi 0
        if(infoStart) infoStart.innerText = 0; 
        if(infoEnd) infoEnd.innerText = 0; 
        if(infoTotal) infoTotal.innerText = 0;
        if(pageInd) pageInd.innerText = "Page 0";
        
        // Matikan Tombol
        if(btnPrev) btnPrev.disabled = true;
        if(btnNext) btnNext.disabled = true;
        return;
    }

    // --- STEP E: SLICING DATA (AMBIL DATA SESUAI HALAMAN) ---
    const startIndex = (currentPage - 1) * limitVal;
    const endIndex = Math.min(startIndex + limitVal, totalData);
    const displayData = filtered.slice(startIndex, endIndex);

    // --- STEP F: RENDER TABEL (LOOPING) ---
    tbody.innerHTML = '';
    
    displayData.forEach(item => {
        // Hitung Indentasi (Berdasarkan jumlah titik di kode)
        const dots = (item.kode.match(/\./g) || []).length;
        const isMain = dots === 0; // Level utama (misal 100, 200)
        const pad = dots * 24 + 16; // Padding kiri dinamis
        
        tbody.innerHTML += `
        <tr class="border-b border-slate-100 hover:bg-slate-50 transition group">
            <td class="px-6 py-3 w-32 font-mono text-sm ${isMain ? 'text-blue-700 font-bold' : 'text-slate-600'} align-top">
                ${item.kode}
            </td>
            <td class="py-3 text-sm pr-6">
                <div style="padding-left: ${pad}px" class="flex items-center gap-2">
                    ${isMain ? '<i class="ph ph-caret-right text-xs text-slate-300"></i>' : '<i class="ph ph-dot text-slate-300 text-xs"></i>'}
                    <span class="${isMain ? 'uppercase font-bold tracking-wide text-slate-800' : 'text-slate-600'}">
                        ${item.deskripsi}
                    </span>
                </div>
            </td>
            <td class="px-6 py-3 text-right w-32">
                <div class="opacity-0 group-hover:opacity-100 flex justify-end gap-2 transition-opacity">
                    <button onclick='editKlasifikasi(${JSON.stringify(item)})' class="w-8 h-8 rounded-full hover:bg-blue-50 text-slate-400 hover:text-blue-600 flex items-center justify-center transition-colors" title="Edit">
                        <i class="ph ph-pencil-simple text-lg"></i>
                    </button>
                    <button onclick="delKlasifikasi('${item.id}')" class="w-8 h-8 rounded-full hover:bg-red-50 text-slate-400 hover:text-red-600 flex items-center justify-center transition-colors" title="Hapus">
                        <i class="ph ph-trash text-lg"></i>
                    </button>
                </div>
            </td>
        </tr>`;
    });

    // --- STEP G: UPDATE FOOTER INFO & BUTTONS ---
    if(infoStart) infoStart.innerText = startIndex + 1;
    if(infoEnd) infoEnd.innerText = endIndex;
    if(infoTotal) infoTotal.innerText = totalData;
    if(pageInd) pageInd.innerText = `Page ${currentPage} / ${totalPages}`;

    // Atur Status Disabled pada Tombol
    if(btnPrev) {
        btnPrev.disabled = (currentPage === 1); // Mati jika di halaman 1
        // Visual feedback (opacity) ditangani oleh class Tailwind 'disabled:opacity-50'
    }
    if(btnNext) {
        btnNext.disabled = (currentPage === totalPages); // Mati jika di halaman terakhir
    }
}

/* --- TAMBAHAN FITUR IMPORT EXCEL --- */

function processImportExcel(input) {
    const file = input.files[0];
    if (!file) return;

    // Tampilkan loading sederhana (ubah text tombol import sementara)
    const btnImport = input.nextElementSibling; // Tombol Import ada setelah input
    const originalText = btnImport.innerHTML;
    btnImport.innerHTML = `<div class="loader"></div> Memproses...`;
    btnImport.disabled = true;

    const reader = new FileReader();

    reader.onload = function(e) {
        try {
            const data = new Uint8Array(e.target.result);
            const workbook = XLSX.read(data, { type: 'array' });

            // Ambil sheet pertama
            const firstSheetName = workbook.SheetNames[0];
            const worksheet = workbook.Sheets[firstSheetName];

            // Konversi ke JSON (Array of Arrays)
            // header: 1 artinya kita ambil raw data baris demi baris
            const jsonData = XLSX.utils.sheet_to_json(worksheet, { header: 1 });

            // Validasi Data
            if (jsonData.length < 2) {
                throw new Error("File Excel kosong atau format salah.");
            }

            // Cleaning Data
            // Asumsi: Baris 1 adalah Header, Data mulai baris 2
            // Kolom A (index 0) = Kode, Kolom B (index 1) = Deskripsi
            let cleanData = [];
            
            // Loop mulai index 1 (melewati header)
            for (let i = 1; i < jsonData.length; i++) {
                const row = jsonData[i];
                // Pastikan ada Kode dan Deskripsi (tidak null/undefined)
                if (row[0] && row[1]) {
                    cleanData.push({
                        kode: String(row[0]).trim(),
                        deskripsi: String(row[1]).trim()
                    });
                }
            }

            if (cleanData.length === 0) {
                throw new Error("Tidak ada data valid yang ditemukan.");
            }

            // Kirim ke Backend GAS
            google.script.run
                .withSuccessHandler((res) => {
                    showToast(`${res.count} Data berhasil diimport!`);
                    loadKlasifikasi(); // Refresh tabel
                    
                    // Reset tombol & input
                    input.value = ""; 
                    btnImport.innerHTML = originalText;
                    btnImport.disabled = false;
                })
                .withFailureHandler((err) => {
                    alert("Gagal Import: " + err);
                    input.value = "";
                    btnImport.innerHTML = originalText;
                    btnImport.disabled = false;
                })
                .importKlasifikasiBulk(cleanData);

        } catch (err) {
            alert("Error: " + err.message);
            input.value = "";
            btnImport.innerHTML = originalText;
            btnImport.disabled = false;
        }
    };

    reader.readAsArrayBuffer(file);
}

function createRow(row, idx, isAktif) {

    if (row.Tahun && String(row.Tahun).trim() !== "") {
        // Split string berdasarkan tanda strip (-)
        const parts = String(row.Tahun).split('-');
        
        if (parts.length === 3) {
            // parts[0] = Tahun, parts[1] = Bulan, parts[2] = Tanggal
            // Susun ulang menjadi: Tanggal/Bulan/Tahun
            dateDisplay = `${parts[2]}/${parts[1]}/${parts[0]}`;

            // Jika format data lama (misalnya cuma "2024"), tampilkan apa adanya
            dateDisplay = row.Tahun;

        function filterDashboard() {
            const search = document.getElementById('dashSearch').value.toLowerCase();
            const category = document.getElementById('dashCategory').value; 
            const start = document.getElementById('dashDateStart').value;
            const end = document.getElementById('dashDateEnd').value;

            const filtered = dashboardDataList.filter(item => {
                // Search Logic
                const matchSearch = (item.uraian || '').toLowerCase().includes(search) || 
                                    (item.noBerkas || '').toLowerCase().includes(search);
                
                // Category Logic
                const itemStatus = (item.status || '').trim();
                const matchCat = category === "" ? true : itemStatus === category;
                
                // --- PERBAIKAN DATE LOGIC (MENGGUNAKAN TANGGAL ARSIP) ---
                let matchDate = true;
                // Gunakan item.tahun (Tanggal Arsip), BUKAN item.tglInput
                if (start && end && item.tahun) {
                    const itemDate = new Date(item.tahun).setHours(0,0,0,0);
                    const startDate = new Date(start).setHours(0,0,0,0);
                    const endDate = new Date(end).setHours(0,0,0,0);
                    
                    // Cek apakah tanggal arsip masuk dalam range
                    matchDate = itemDate >= startDate && itemDate <= endDate;
                } else if (start && end && !item.tahun) {
                    // Jika user filter tanggal, tapi data arsipnya tidak punya tanggal arsip -> Hide
                    matchDate = false;
                }

                return matchSearch && matchCat && matchDate;
            });

            renderDashboardGrid(filtered);
        }

        loc3: document.getElementById('inpLoc3').value,

        oldFileId: document.getElementById('inpOldFileId').value,
        oldFileUrl: document.getElementById('inpOldFileUrl').value
    };

    // Fungsi Pengiriman ke Backend
    const send = (data) => {
        google.script.run.withSuccessHandler(res => {
            if(res.success) {
                showToast(res.message);
                closeModal('modalArsip');
                loadDashboard();
                loadArsip('Aktif');
                loadArsip('Inaktif');
            } else {
                // Jika Token Salah / Error Server
                Swal.fire('Gagal!', res.message, 'error');
                if(res.message.includes("Login") || res.message.includes("Token")) {
                    logout(); // Logout otomatis jika token kadaluarsa
                }
            }
            // Reset UI
            btn.disabled = false;

        })
        .withFailureHandler(err => {
             Swal.fire('Error Sistem', err.message, 'error');
             btn.disabled = false;
             loader.classList.add('hidden');
        })
        .saveArsip(data);
    };

    // Logika Upload File (FileReader)
    if (file) {
        const reader = new FileReader();
        reader.onload = function(ev) {
            formData.fileData = ev.target.result.split(',')[1]; // Base64
            formData.fileName = file.name;
            formData.mimeType = file.type;
            send(formData);
        }
        reader.readAsDataURL(file);
    } else {
        // Jika tidak ada file baru yang diupload
        send(formData);
    }
}
function delArsip(id, fileId) {
    Swal.fire({
        title: 'Yakin hapus permanen?',
        text: "Data arsip yang dihapus tidak dapat dikembalikan lagi! (Butuh Akses Admin)",
        icon: 'warning',
        showCancelButton: true,
        confirmButtonColor: '#d33',
        cancelButtonColor: '#3085d6',

        reverseButtons: true,
        customClass: {
            popup: 'rounded-2xl',
            confirmButton: 'rounded-xl px-6',
            cancelButton: 'rounded-xl px-6'
        }
    }).then((result) => {
        if (result.isConfirmed) {
            
            // 1. AMBIL TOKEN DARI STORAGE
            const token = localStorage.getItem('sispardi_token');

            // Cek Sisi Klien (Mencegah request sia-sia)
            if(!token) {
                Swal.fire('Akses Ditolak', 'Sesi login tidak valid. Silakan login ulang.', 'error');
                return;
            }

            Swal.fire({ title: 'Sedang menghapus...', didOpen: () => Swal.showLoading() });

            google.script.run
                .withSuccessHandler((res) => {
                    if(res.success) {
                        Swal.fire({
                            title: 'Terhapus!',
                            text: res.message,
                            icon: 'success',
                            customClass: { popup: 'rounded-2xl', confirmButton: 'rounded-xl px-6' }
                        });
                        loadDashboard();
                        loadArsip('Aktif');
                        loadArsip('Inaktif');
                    } else {
                        // Error dari Backend (Token Salah / Bukan Admin)
                        Swal.fire({
                            title: 'Gagal!',
                            text: res.message,
                            icon: 'error',
                            customClass: { popup: 'rounded-2xl', confirmButton: 'rounded-xl px-6' }
                        });
                        
                        // Jika sesi habis/token invalid, logout otomatis
                        if(res.message.includes("Login") || res.message.includes("Token")) {
                            logout();
                        }

                })

                    Swal.fire('Error Sistem', err.toString(), 'error');
                })

        }
    });

function parseTahun(val) {
    if (!val) return "";
    // Jika tipe datanya sudah angka, langsung kembalikan
    if (typeof val === 'number') return val;
    
    // Jika string, cari angka pertama
    const match = String(val).match(/\d+/);
    return match ? parseInt(match[0]) : "";
}

function autofillKlasifikasi() {
    const inputKode = document.getElementById('inpKode').value;
    
    // 1. Cari data yang cocok di variabel global rawKlasifikasiData
    const found = rawKlasifikasiData.find(item => 
        String(item.kode).trim().toLowerCase() === String(inputKode).trim().toLowerCase()
    );

    if (found) {
        // --- [BARU] AUTOFILL URAIAN BERKAS ---
        // Mengisi Uraian Berkas dengan Deskripsi dari Klasifikasi
        // Jika deskripsi ada, masukkan ke kolom inpUraian
        if (found.deskripsi) {
            document.getElementById('inpUraian').value = found.deskripsi;
        }

        // --- AUTOFILL RETENSI & KARAKTERISTIK (LAMA) ---
        
        // A. RETENSI AKTIF
        if (found.retensiAktif) {
            document.getElementById('inpRA').value = parseTahun(found.retensiAktif);
        }

        // B. RETENSI INAKTIF
        if (found.retensiInaktif) {
            document.getElementById('inpRI').value = parseTahun(found.retensiInaktif);
        }

        // C. KETERANGAN / NASIB AKHIR
        if (found.keterangan) {
            const ketVal = String(found.keterangan).trim();
            const elmKet = document.getElementById('inpKeterangan');
            
            for (let i = 0; i < elmKet.options.length; i++) {
                if (elmKet.options[i].value.toLowerCase() === ketVal.toLowerCase()) {
                    elmKet.selectedIndex = i;
                    break;
                }
            }
        }

        // D. KEAMANAN
        if (found.keamanan) {
            const secVal = String(found.keamanan).trim();
            const elmSec = document.getElementById('inpKeamanan');
            
            for (let i = 0; i < elmSec.options.length; i++) {
                if (elmSec.options[i].value.toLowerCase() === secVal.toLowerCase()) {
                    elmSec.selectedIndex = i;
                    break;
                }
            }
        }
        
        // Cek status otomatis setelah angka berubah
        cekStatusOtomatis();

    } else {
        // Jika kode tidak ditemukan atau dihapus user, biarkan isian tetap seperti adanya
        // agar data manual user tidak hilang tiba-tiba.
    }
}

      function handleAutoNumbering(e) {

                  const val = input.value;

                  const end = input.selectionEnd;

                  // Ambil baris saat ini (di posisi kursor)

                  // Cek apakah baris saat ini dimulai dengan angka (contoh: "1. ")
                  const match = currentLine.match(/^(\d+)\.\s/);

                  if (match) {
                      e.preventDefault(); // Mencegah enter standar
                      const currentNumber = parseInt(match[1]);
                      const nextNumber = currentNumber + 1;
                      const insertion = `\n${nextNumber}. `;

                      // Sisipkan nomor baru di posisi kursor
                      input.value = val.substring(0, start) + insertion + val.substring(end);

                      // Pindahkan kursor ke setelah nomor baru
                      input.selectionStart = input.selectionEnd = start + insertion.length;
                  } else if (lines.length === 1 && currentLine.length > 0 && !currentLine.match(/^\d+\./)) {
                      // Opsional: Jika baris pertama belum ada nomor dan di-enter, 
                      // bisa otomatis jadi no 2, tapi sebaiknya biarkan user mengetik "1. " di awal manual
                      // agar fleksibel.
                  }
              }
          }

function performBackup() {
    const token = localStorage.getItem('sispardi_token');
    if(!token) return logout();

    Swal.fire({
        title: 'Mencadangkan Database...',
        html: 'Mengunduh seluruh data arsip dari server.',
        allowOutsideClick: false,
        didOpen: () => Swal.showLoading()
    });

    google.script.run
        .withSuccessHandler(data => {
            if(!data || data.length === 0) {
                Swal.fire('Info', 'Database masih kosong.', 'info');
                return;
            }

            const dataToExport = data.map((row, index) => {
                return {
                    "No": index + 1,
                    "Bidang": row.Bidang || '-',
                    "Kode Klasifikasi": row.KodeKlasifikasi,
                    "Indeks": row.Indeks || '-',
                    "No. Berkas": row.NoBerkas,
                    "Uraian Berkas": row.Uraian,
                    "Uraian Arsip": row.UraianArsip || '-',
                    "Tanggal Arsip": row.Tahun || '-',
                    "Jml": row.Jumlah || '1',
                    "Tingkat Perkemb.": row.TingkatPerkembangan,
                    "Keamanan": row.Keamanan,
                    "RA": row.RetensiAktif,
                    "RI": row.RetensiInaktif,
                    "Status": row.Status || '-',
                    "Ket.": row.Ket || '-',
                    "No. Filling Cabinet / Box": row.Loc1 || '-',
                    "No. Laci / Rak": row.Loc2 || '-',
                    "No. Folder": row.Loc3 || '-'
                };
            });

            const worksheet = XLSX.utils.json_to_sheet(dataToExport);
            
            // Auto Width
            const wscols = [
                {wch: 5},  {wch: 15}, {wch: 15}, {wch: 20}, {wch: 20},
                {wch: 40}, {wch: 40}, {wch: 15}, {wch: 5},  {wch: 15},
                {wch: 10}, {wch: 5},  {wch: 5},  {wch: 10}, {wch: 15},
                {wch: 20}, {wch: 15}, {wch: 10}
            ];
            worksheet['!cols'] = wscols;

            const workbook = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(workbook, worksheet, "Backup_Arsip");

            const today = new Date();
            const dateStr = today.toISOString().split('T')[0];
            XLSX.writeFile(workbook, `Backup_Database_sispardi_${dateStr}.xlsx`);

            Swal.close();
            showToast("Backup berhasil diunduh");
        })
        .withFailureHandler(err => {
            Swal.fire('Error', err.message, 'error');
            if(err.message.includes("Sesi") || err.message.includes("Token")) {
                logout();
            }
        })
        .backupDatabase(token);
}

document.addEventListener('DOMContentLoaded', function() {
    
    // Target: Textarea Uraian Arsip
    const inpArsip = document.getElementById('inpUraianArsip');
    
    if (inpArsip) {
        // Placeholder agar user tahu fiturnya
        inpArsip.placeholder = "Ketik rincian arsip di sini...\n(Tekan Enter untuk otomatis membuat nomor urut)";

        inpArsip.addEventListener('keydown', function(e) {
            if (e.key === 'Enter') {
                const input = e.target;
                const val = input.value;
                const start = input.selectionStart;
                const end = input.selectionEnd;

                // Ambil teks sampai posisi kursor
                const textUpToCursor = val.substring(0, start);
                const lines = textUpToCursor.split('\n');
                const currentLineIndex = lines.length - 1;
                const currentLine = lines[currentLineIndex];

                // CEK 1: Apakah baris ini sudah ada nomornya? (Contoh: "1. Surat")
                const matchNumber = currentLine.match(/^(\d+)\.\s/);

                if (matchNumber) {
                    // KASUS A: Sudah ada nomor -> Lanjutkan nomor berikutnya
                    e.preventDefault();
                    const currentNumber = parseInt(matchNumber[1]);
                    const nextNumber = currentNumber + 1;
                    const insertion = `\n${nextNumber}. `;

                    input.value = val.substring(0, start) + insertion + val.substring(end);
                    input.selectionStart = input.selectionEnd = start + insertion.length;

                } else if (currentLineIndex === 0 && currentLine.trim().length > 0) {
                    // KASUS B (BARU): Baris pertama, ada teks, tapi BELUM ada nomor
                    // User lupa ketik "1.", jadi kita buatkan otomatis.
                    e.preventDefault();

                    const textAfterCursor = val.substring(end);
                    
                    // Format baru: "1. " + Teks Lama + "\n2. " + Teks Sisa
                    const newFirstLine = "1. " + currentLine; 
                    const insertion = "\n2. ";
                    
                    input.value = newFirstLine + insertion + textAfterCursor;
                    
                    // Pindahkan kursor ke setelah "2. "
                    const newCursorPos = newFirstLine.length + insertion.length;
                    input.selectionStart = input.selectionEnd = newCursorPos;
                }
                // Jika tidak masuk kasus A atau B, Enter berfungsi normal (baris baru biasa)
            }
        });
    }
});
